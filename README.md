### cpu

````python
def cpu_stress():
    try:
        global time_stress
        old_time = time.time()+ float(time_stress)
        while old_time > time.time():
            processes = cpu_count()
            pool = Pool(processes)
            pool.map(cpu_time, range(processes))
            if time.time() > old_time:
                break
    except Exception as err:
        windows_txt(err)
    
def cpu_time(x):
    try:
        global time_stress
        old_time = time.time()+ float(time_stress)
        while old_time > time.time():
            x*x
    except Exception as err:
        windows_txt(err)
````
